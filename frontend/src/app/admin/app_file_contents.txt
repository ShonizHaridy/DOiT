import os

def write_files_to_txt(directory, output_file, ignored_directories, files_only_ignored_directories, ignored_files, ignored_extensions):
    normalized_ignored_dirs = [os.path.normpath(os.path.abspath(d)) for d in ignored_directories]
    normalized_files_only_ignored_dirs = [os.path.normpath(os.path.abspath(d)) for d in files_only_ignored_directories]

    with open(output_file, "w", encoding="utf-8") as out:
        for root, dirs, files in os.walk(directory):
            dirs_to_remove = []
            for d in dirs:
                dir_path = os.path.normpath(os.path.abspath(os.path.join(root, d)))
                if dir_path in normalized_ignored_dirs:
                    dirs_to_remove.append(d)
            for d in dirs_to_remove:
                dirs.remove(d)

            for file in files:
                file_path = os.path.join(root, file)
                ext = os.path.splitext(file)[1].lower()

                if (file_path in ignored_files or file in ignored_files) or (ext in ignored_extensions):
                    continue

                skip_file = False
                for files_only_ignored_dir in normalized_files_only_ignored_dirs:
                    if os.path.normpath(os.path.abspath(root)) == files_only_ignored_dir:
                        skip_file = True
                        break
                if skip_file:
                    continue

                out.write(f"File Path: {file_path}\n")
                try:
                    with open(file_path, "r", encoding="utf-8") as f:
                        content = f.read()
                        out.write("Content:\n")
                        out.write(content + "\n")
                except Exception as e:
                    out.write(f"Could not read file {file_path}: {e}\n")

                out.write("\n" + "-" * 80 + "\n\n")

def get_ignored_extensions():
    exts = []
    count = input("Enter number of file extensions to ignore (0 for none): ")

        count = int(count)
        for i in range(count):
